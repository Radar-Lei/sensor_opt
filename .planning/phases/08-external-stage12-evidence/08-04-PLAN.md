    if key not in gate:

if int(gate['required_split_count']) != 10:
    raise SystemExit(f'expected required_split_count 10, got {gate["required_split_count"]}')
for dataset, info in gate.get('datasets', {}).items():
    status = info.get('evidence_status')
    sources = info.get('required_sources') or info.get('source_csvs') or []
    if status == 'complete_stage12_10_split':
        for source in sources:
            if isinstance(source, dict):
                source = source.get('path') or source.get('source_csv') or ''
            if not source:
                continue
            result = subprocess.run(['git', 'ls-files', '--error-unmatch', str(source)], stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL, check=False)
            if result.returncode != 0:
                raise SystemExit(f'completed dataset {dataset} has untracked source: {source}')
    if status == 'pending_tracking' and info.get('complete'):
        raise SystemExit(f'pending_tracking dataset {dataset} cannot be complete')
if gate['seattle_stage12_complete'] is False and gate['seattle_core_claim_blocked'] is not True:
    raise SystemExit('incomplete Seattle evidence must block core claims')
if gate['seattle_core_claim_blocked'] is False and gate['seattle_stage12_complete'] is not True:
    raise SystemExit('unblocked Seattle requires complete Stage12 evidence')
